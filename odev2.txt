Aşağıdaki kodlardan herbirinde function overloading ("işlev bildirimi") olup olmadığını belirtin.
Her bir kod için açıklama yapın:

//-----------------------------------------------------
//-----------------------------------------------------

void func(int);

void foo()
{
	extern func(int, int);
}
//-----------------------------------------------------
//-----------------------------------------------------
int func(double);
double func(double);
//-----------------------------------------------------
//-----------------------------------------------------
int f(double);
int f(double = 0.);
//-----------------------------------------------------
//-----------------------------------------------------
int g(int);
int g(const int);
//-----------------------------------------------------
//-----------------------------------------------------
int g(int *p);
int g(int* const p);
//-----------------------------------------------------
//-----------------------------------------------------
int f(int *p);
int f(const int *p);
//-----------------------------------------------------
//-----------------------------------------------------
void func(int &);
void func(const int &);
//-----------------------------------------------------
//-----------------------------------------------------
void g(int);
void g(int &);
//-----------------------------------------------------
//-----------------------------------------------------
void foo(int &);
void foo(int &&);
//-----------------------------------------------------
//-----------------------------------------------------
void f(const double &);
void f(double &&);
//-----------------------------------------------------
//-----------------------------------------------------
